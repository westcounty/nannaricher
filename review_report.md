  // server/src/game/rules/WinConditions.ts

  export class WinConditionChecker {

    // 检查所有胜利条件
    checkAllWinConditions(player: Player, state: GameState): WinResult | null {
      // 1. 基础胜利条件
      const baseResult = this.checkBaseWinCondition(player);
      if (baseResult) return baseResult;

      // 2. 培养计划胜利条件
      for (const planId of player.confirmedPlans) {
        const planResult = this.checkPlanWinCondition(player, planId, state);
        if (planResult) return planResult;
      }

      return null;
    }

    // 基础胜利：GPA*10 + 探索值 >= 60
    private checkBaseWinCondition(player: Player): WinResult | null {
      const score = player.gpa * 10 + player.exploration;
      if (score >= 60) {
        return { condition: '基础胜利', detail: `GPA*10+探索值=${score.toFixed(1)}≥60` };
      }
      return null;
    }

    // 33种培养计划胜利条件
    private checkPlanWinCondition(
      player: Player,
      planId: string,
      state: GameState
    ): WinResult | null {

      switch (planId) {
        // ===== 文学院 =====
        case 'plan_wenxue': {
          // 离开赚在南哪线时，金钱未发生变化
          const lastExit = this.getLastLineExit(player, 'money');
          if (lastExit && lastExit.moneyBefore === lastExit.moneyAfter) {
            return { condition: '文学院', detail: '离开赚在南哪线时金钱未变化' };
          }
          break;
        }

        // ===== 历史学院 =====
        case 'plan_lishi': {
          // 按顺序经过鼓楼、浦口、仙林、苏州
          const order = ['gulou', 'pukou', 'xianlin', 'suzhou'];
          const visited = player.history.lineEventsTriggered;
          let inOrder = true;
          let lastIdx = -1;
          for (const line of order) {
            const lineVisits = visited[line] || [];
            if (lineVisits.length === 0) { inOrder = false; break; }
            const firstVisit = Math.min(...lineVisits);
            if (firstVisit < lastIdx) { inOrder = false; break; }
            lastIdx = firstVisit;
          }
          if (inOrder) {
            return { condition: '历史学院', detail: '按顺序经过四个校区' };
          }
          break;
        }

        // ===== 哲学系 =====
        case 'plan_zhexue': {
          // 完整进出某条线，GPA和探索值无变化
          for (const exit of player.history.lineExits) {
            if (exit.gpaBefore === exit.gpaAfter &&
                exit.explorationBefore === exit.explorationAfter) {
              return { condition: '哲学系', detail: '进出线路无属性变化' };
            }
          }
          break;
        }

        // ===== 法学院 =====
        case 'plan_faxue': {
          // 场上出现破产玩家且不是自己
          const hasOtherBankrupt = state.players.some(
            p => p.id !== player.id && p.isBankrupt
          );
          if (hasOtherBankrupt) {
            return { condition: '法学院', detail: '场上出现破产玩家' };
          }
          break;
        }

        // ===== 商学院 =====
        case 'plan_shangxue': {
          if (player.money >= 5000) {
            return { condition: '商学院', detail: `金钱达到${player.money}` };
          }
          break;
        }

        // ===== 外国语学院 =====
        case 'plan_waiguoyu': {
          // 抽到过两张包含英文字母的卡（除GPA外）
          const englishCards = player.history.cardsDrawn.filter(c => c.hasEnglish);
          if (englishCards.length >= 2) {
            return { condition: '外国语学院', detail: `抽取${englishCards.length}张含英文卡` };
          }
          break;
        }

        // ===== 新闻传播学院 =====
        case 'plan_xinwen': {
          // 完整经过乐在南哪线，无探索值和GPA扣减
          const exploreExits = player.history.lineExits.filter(e => e.lineId === 'explore');
          for (const exit of exploreExits) {
            if (exit.explorationBefore <= exit.explorationAfter &&
                exit.gpaBefore <= exit.gpaAfter) {
              return { condition: '新闻传播学院', detail: '乐在南哪线无扣减' };
            }
          }
          break;
        }

        // ===== 政府管理学院 =====
        case 'plan_zhengguan': {
          // 三项属性均不与其他玩家一致
          const others = state.players.filter(p => p.id !== player.id);
          const noMatch = others.every(p =>
            p.money !== player.money &&
            p.gpa !== player.gpa &&
            p.exploration !== player.exploration
          );
          if (noMatch && others.length > 0) {
            return { condition: '政府管理学院', detail: '属性均与其他玩家不同' };
          }
          break;
        }

        // ===== 国际关系学院 =====
        case 'plan_guoji': {
          // 和至少两名其他玩家互相使用过机会卡
          const usage = player.history.chanceCardsUsedOnPlayers;
          const mutualCount = Object.entries(usage).filter(([targetId, count]) => {
            const target = state.players.find(p => p.id === targetId);
            if (!target) return false;
            const reverseCount = target.history.chanceCardsUsedOnPlayers[player.id] || 0;
            return count >= 1 && reverseCount >= 1;
          }).length;
          if (mutualCount >= 2) {
            return { condition: '国际关系学院', detail: `与${mutualCount}人互相使用机会卡` };
          }
          break;
        }

        // ===== 信息管理学院 =====
        case 'plan_xinxiguanli': {
          // 抽到过5张不重复的数字开头卡
          const digitCards = new Set(
            player.history.cardsDrawn
              .filter(c => c.startsWithWithDigit)
              .map(c => c.cardId)
          );
          if (digitCards.size >= 5) {
            return { condition: '信息管理学院', detail: `${digitCards.size}张数字开头卡` };
          }
          break;
        }

        // ===== 社会学院 =====
        case 'plan_shehuixue': {
          const others = state.players.filter(p => p.id !== player.id);
          if (others.length > 0) {
            const lowestExp = Math.min(...others.map(p => p.exploration));
            if (player.exploration >= lowestExp + 20) {
              return { condition: '社会学院', detail: `探索值比最低高${player.exploration - lowestExp}` };
            }
          }
          break;
        }

        // ===== 数学系 =====
        case 'plan_shuxue': {
          // 第三次到达鼓楼线终点
          const gulouExits = player.history.lineExits.filter(e => e.lineId === 'gulou');
          if (gulouExits.length >= 3) {
            return { condition: '数学系', detail: `${gulouExits.length}次到达鼓楼线终点` };
          }
          break;
        }

        // ===== 物理学院 =====
        case 'plan_wuli': {
          // 任选两项达到60
          const scores = [
            player.exploration,
            player.gpa * 10,
            player.money / 100
          ];
          let validPairs = 0;
          for (let i = 0; i < scores.length; i++) {
            for (let j = i + 1; j < scores.length; j++) {
              if (scores[i] >= 60 && scores[j] >= 60) validPairs++;
            }
          }
          if (validPairs >= 1) {
            return { condition: '物理学院', detail: '两项属性分数均≥60' };
          }
          break;
        }

        // ===== 天文与空间科学学院 =====
        case 'plan_tianwen': {
          // 与所有其他玩家同时在同一格子停留过
          const others = state.players.filter(p => p.id !== player.id);
          const allMet = others.every(other => {
            const sharedCells = player.history.sharedCellsWith[other.id] || [];
            return sharedCells.length > 0;
          });
          if (allMet && others.length > 0) {
            return { condition: '天文学院', detail: '与所有玩家在同一格相遇' };
          }
          break;
        }

        // ===== 化学化工学院 =====
        case 'plan_huaxue': {
          if (player.exploration >= 45) {
            return { condition: '化学化工学院', detail: `探索值达到${player.exploration}` };
          }
          break;
        }

        // ===== 人工智能学院 =====
        case 'plan_rengong': {
          const others = state.players.filter(p => p.id !== player.id);
          if (others.length > 0) {
            const lowestGpa = Math.min(...others.map(p => p.gpa));
            if (player.gpa >= lowestGpa + 2.0) {
              return { condition: '人工智能学院', detail: `GPA比最低高${(player.gpa - lowestGpa).toFixed(1)}` };
            }
          }
          break;
        }

        // ===== 计算机科学与技术系 =====
        case 'plan_jisuanji': {
          // 探索值和金钱数字只包含0或1
          const expStr = Math.floor(player.exploration).toString();
          const moneyStr = player.money.toString();
          const only01 = (s: string) => /^[01]+$/.test(s);
          if (only01(expStr) && only01(moneyStr)) {
            return { condition: '计算机系', detail: `探索值${expStr} 金钱${moneyStr}` };
          }
          break;
        }

        // ===== 软件学院 =====
        case 'plan_ruanjian': {
          // 到达交学费格，支出3200未破产
          // 需要在事件处理中追踪
          break;
        }

        // ===== 电子科学与工程学院 =====
        case 'plan_dianzi': {
          // 在科创赛事投到6
          // 需要在事件处理中标记
          if (player.stats.kechuangRolledSix) {
            return { condition: '电子学院', detail: '科创赛事投到6' };
          }
          break;
        }

        // ===== 现代工程与应用科学学院 =====
        case 'plan_xiandai': {
          // 进入过除苏州外所有线
          const required = ['pukou', 'study', 'money', 'explore', 'gulou', 'xianlin', 'food'];
          const allEntered = required.every(line =>
            player.history.linesVisited.includes(line)
          );
          if (allEntered) {
            return { condition: '现代工程学院', detail: '进入所有线路（除苏州）' };
          }
          break;
        }

        // ===== 环境学院 =====
        case 'plan_huanjing': {
          // 经历过仙林线每一个事件
          const xianlinEvents = player.history.lineEventsTriggered['xianlin'] || [];
          if (xianlinEvents.length >= 7) { // 仙林线7格
            return { condition: '环境学院', detail: '经历仙林线所有事件' };
          }
          break;

        // ===== 地球科学与工程学院 =====
        case 'plan_diqiu': {
          // 进入过每一条线
          const allLines = ['pukou', 'study', 'money', 'suzhou', 'explore', 'gulou', 'xianlin', 'food'];
          const allEntered = allLines.every(line =>
            player.history.linesVisited.includes(line)
          );
          if (allEntered) {
            return { condition: '地球科学学院', detail: '进入所有8条线路' };
          }
          break;

        case 'plan_dili': {

            return { condition: '地理学院', detail: '完成四个校区线' };
          }
          break;

        // ===== 大气科学学院 =====
        case 'plan_daqi': {

          if (state.turnNumber >= 20) {
            const moneyHistory = player.history.moneyHistory.slice(-20);

              const othersAtTurn = state.players
                .filter(p => p.id !== player.id)
                .map(p => p.history.moneyHistory[idx]);
              const maxOther = Math.max(...othersAtTurn);
              return money < maxOther;

  // client/src/audio/AudioManager.ts

  import { Howl, Howler } from 'howler';

  export class AudioManager {
    private static instance: AudioManager;
    private sounds: Map<string, Howl> = new Map();
    private music: Howl | null = null;
    private volume = {
      master: 1.0,
      bgm: 0.5,
      sfx: 0.7,
      ui: 0.6,
    };
    private muted = false;

    static getInstance(): AudioManager {
      if (!AudioManager.instance) {
        AudioManager.instance = new AudioManager();
      }
      return AudioManager.instance;
    }

    // 初始化 - 预加载所有音效
    async initialize(): Promise<void> {
      // 预加载关键音效
      const criticalSounds = [
        'dice/roll', 'dice/bounce',
        'piece/step', 'piece/arrive',
        'card/draw', 'card/flip',
        'resource/money_gain', 'resource/money_loss',
        'game/turn_start', 'victory/win',
      ];

      await Promise.all(
        criticalSounds.map(id => this.load(id))
      );

      // 设置全局音量
      Howler.volume(this.volume.master);
    }

    // 加载音效
    private async load(soundId: string): Promise<void> {
      return new Promise((resolve, reject) => {
        const sound = new Howl({
          src: [`/audio/${soundId}.mp3`],
          preload: true,
          onload: () => resolve(),
          onloaderror: (_, error) => reject(error),
        });
        this.sounds.set(soundId, sound);
      });
    }

    // 播放音效
    play(soundId: string, options: PlayOptions = {}): void {
      if (this.muted) return;

      const sound = this.sounds.get(soundId);
      if (!sound) {
        console.warn(`Sound not found: ${soundId}`);
        // 动态加载
        this.load(soundId).then(() => this.play(soundId, options));
        return;
      }

      const vol = this.getVolumeForCategory(options.category || 'sfx');
      sound.volume(vol * (options.volume || 1));
      sound.rate(options.rate || 1);
      sound.play();
    }

    // 播放BGM
    playMusic(musicId: string, options: MusicOptions = {}): void {
      // 淡出当前BGM
      if (this.music) {
        this.music.fade(this.music.volume(), 0, 1000);
        setTimeout(() => this.music?.stop(), 1000);
      }

      // 播放新BGM
      this.music = new Howl({
        src: [`/audio/music/${musicId}.mp3`],
        loop: options.loop !== false,
        volume: 0,
      });

      this.music.play();
      this.music.fade(0, this.volume.bgm, options.fadeIn || 2000);
    }

    // 停止BGM
    stopMusic(fadeOut: number = 1000): void {
      if (this.music) {
        this.music.fade(this.music.volume(), 0, fadeOut);
        setTimeout(() => {
          this.music?.stop();
          this.music = null;
        }, fadeOut);
      }
    }

    // 设置音量
    setVolume(type: 'master' | 'bgm' | 'sfx' | 'ui', value: number): void {
      this.volume[type] = Math.max(0, Math.min(1, value));

      if (type === 'master') {
        Howler.volume(this.volume.master);
      } else if (type === 'bgm' && this.music) {
        this.music.volume(this.volume.bgm);
      }
    }

    // 静音切换
    toggleMute(): boolean {
      this.muted = !this.muted;
      Howler.mute(this.muted);
      return this.muted;
    }

    // 获取分类音量
    private getVolumeForCategory(category: 'sfx' | 'ui' | 'ambient'): number {
      switch (category) {
        case 'ui': return this.volume.ui;
        case 'ambient': return this.volume.sfx * 0.5;
        default: return this.volume.sfx;
      }
    }
  }

  // AccessibleModal.tsx
  export function AccessibleModal({
    isOpen,
    onClose,
    title,
    children
  }: ModalProps) {
    const modalRef = useRef<HTMLDivElement>(null);
    const previousFocus = useRef<HTMLElement | null>(null);

    useEffect(() => {
      if (isOpen) {
        // 保存当前焦点
        previousFocus.current = document.activeElement as HTMLElement;
        // 聚焦到模态框
        modalRef.current?.focus();
        // 禁止背景滚动
        document.body.style.overflow = 'hidden';
      } else {
        // 恢复焦点
        previousFocus.current?.focus();
        document.body.style.overflow = '';
      }
    }, [isOpen]);

    // 键盘处理
    const handleKeyDown = (e: React.KeyboardEvent) => {
      if (e.key === 'Escape') {
        onClose();
      }
      // Tab陷阱
      if (e.key === 'Tab') {
        trapFocus(e, modalRef.current);
      }
    };

    if (!isOpen) return null;

    return (
      <div
        className="modal-overlay"
        onClick={onClose}
        role="presentation"
      >
        <div
          ref={modalRef}
          className="modal-content"
          role="dialog"
          aria-modal="true"
          aria-labelledby="modal-title"
          tabIndex={-1}
          onKeyDown={handleKeyDown}
        >
          <h2 id="modal-title" className="sr-only">{title}</h2>
          {children}
        </div>
      </div>
    );
  }
